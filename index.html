        // --- Simulator Logic ---

        const simState = {

            currentStep: 1,

            userInput: "",

            preset: "0",

            batched: "0",

            isBatching: false,

        };

        const presetEl = document.getElementById('preset-value-sim');

        const batchEl = document.getElementById('batch-value-sim');

        function updateSimDisplay() {

            presetEl.textContent = simState.userInput || simState.preset + " kg";

            batchEl.textContent = simState.batched + " kg";

        }

        

        function updateStep(step) {

            simState.currentStep = step;

            document.querySelectorAll('.step-instruction').forEach(el => el.classList.remove('active'));

            const activeStep = document.getElementById(`step-${step}`);

            if (activeStep) activeStep.classList.add('active');

        }


        function handleKeyPress(key) {

            switch(simState.currentStep) {

                case 1: // Waiting for RESET

                    if (key === 'RESET') {

                        simState.preset = "0";

                        simState.batched = "0";

                        simState.userInput = "";

                        updateStep(2);

                    }

                    break;

                case 2: // Waiting for number input

                    if (!isNaN(key)) {

                        simState.userInput += key;

                        if (simState.userInput === simState.targetPreset) {

                            updateStep(3);

                        }

                    } else if (key === 'RESET') { // Allow reset during input

                        simState.userInput = "";

                    }

                    break;

                case 3: // Waiting for Set

                    if (key === 'Set' && simState.userInput === simState.targetPreset) {

                        simState.preset = simState.userInput;

                        simState.userInput = "";

                        updateStep(4);

                    }

                    break;

                case 4: // Waiting for Start

                    if (key === 'Start' && !simState.isBatching) {

                        startBatching();

                    }

                    break;

                case 5: // Batch finished

                     if (key === 'RESET') {

                        resetSimulator();

                    }

                    break;

            }

            if (key === 'Stop' && simState.isBatching) stopBatching();


            updateSimDisplay();

        }

        

        function startBatching() {

            simState.isBatching = true;

            let currentAmount = parseInt(simState.batched);

            const targetAmount = parseInt(simState.preset);


            simState.batchInterval = setInterval(() => {

                currentAmount += Math.round(targetAmount / 20); // Simulate fill speed

                if (currentAmount >= targetAmount) {

                    currentAmount = targetAmount;

                    stopBatching();

                    updateStep(5);

                    progressText.textContent = "¡Dosificación Completada!";

                }

                simState.batched = currentAmount.toString();

                let progress = (currentAmount / targetAmount) * 100;

                progressFill.style.width = `${progress}%`;

                progressText.textContent = `Dosificando... ${currentAmount} / ${targetAmount} kg`;

                updateSimDisplay();

            }, 200);

        }


        function stopBatching() {

            clearInterval(simState.batchInterval);

            simState.isBatching = false;

            if(parseInt(simState.batched) < parseInt(simState.preset)) {

                progressText.textContent = `Pausado en ${simState.batched} kg. Presione START para continuar.`;

            }

        }

        

        function resetSimulator() {

            stopBatching();

            Object.assign(simState, { currentStep: 1, userInput: "", preset: "0", batched: "0" });

            progressFill.style.width = '0%';

            progressText.textContent = 'Esperando inicio...';

            updateStep(1);

            updateSimDisplay();
